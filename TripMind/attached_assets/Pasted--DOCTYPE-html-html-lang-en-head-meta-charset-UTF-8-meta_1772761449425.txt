        async function callGemini(prompt) {
            if (!GEMINI_KEY || GEMINI_KEY.includes('REPLACE_WITH')) {
                throw new Error('Gemini API key is not configured. Please add your key to the script.');
            }
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${GEMINI_KEY}`;
            const response = await fetch(url, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    contents: [{ parts: [{ text: prompt }] }],
                    generationConfig: { temperature: 0.4, responseMimeType: "application/json" }
                })
            });
            const data = await response.json();
            if (!response.ok) throw new Error(data.error?.message || 'Gemini API call failed');
            let text = data.candidates[0].content.parts[0].text;
            text = text.replace(/```json/g, '').replace(/```/g, '').trim();
            return JSON.parse(text);
        }

        const DayPlanner = ({ trip, onUpdate, showToast }) => {
            const [activeDayIdx, setActiveDayIdx] = useState(0);
            const [days, setDays] = useState([]);
            const [loading, setLoading] = useState(true);
            const [showAddForm, setShowAddForm] = useState(false);
            const [editingActivity, setEditingActivity] = useState(null);

            const fetchTripData = useCallback(async () => {
                if (trip.id === 'demo-rajasthan') { setDays(trip.itinerary.days); setLoading(false); return; }
                setLoading(true);
                try {
                    const { data, error } = await supabase.from('days').select('*, activities(*)').eq('trip_id', trip.id).order('day_number', { ascending: true });
                    if (error) throw error;
                    setDays(data.map(day => ({ ...day, activities: (day.activities || []).sort((a, b) => a.sort_order - b.sort_order) })));
                } catch (err) { showToast(err.message, "error"); } finally { setLoading(false); }
            }, [trip, showToast]);

            useEffect(() => { fetchTripData(); }, [fetchTripData]);

            const handleToggle = async (dayId, activity) => {
                if (trip.id === 'demo-rajasthan') return showToast("Demo is read-only", "warning");
                try {
                    const { error } = await supabase.from('activities').update({ status: activity.status === 'done' ? 'planned' : 'done' }).eq('id', activity.id);
                    if (error) throw error; fetchTripData(); onUpdate();
                } catch (err) { showToast(err.message, "error"); }
            };

            const handleDelete = async (activityId) => {
                if (trip.id === 'demo-rajasthan') return showToast("Demo is read-only", "warning");

                try {
                    const { error } = await supabase.from('activities').delete().eq('id', activityId);
                    if (error) throw error; fetchTripData(); onUpdate();
                } catch (err) { showToast(err.message, "error"); }
            };

                e.preventDefault();
                if (trip.id === 'demo-rajasthan') return showToast("Demo is read-only", "warning");

                const data = Object.fromEntries(fd);

                        )}
                    </div>
                    {showAddForm && (
                        <div className="fixed inset-0 z-50 flex items-end sm:items-center justify-center p-4 bg-black/40 backdrop-blur-sm animate-in fade-in duration-200">
                            <div className="w-full max-w-md bg-white rounded-[2rem] p-8 shadow-2xl animate-in slide-in-from-bottom duration-300">

'Edit Activity' : 'Add Activity'}</h3><button onClick={() => setShowAddForm(false)} className="p-2 hover:bg-gray-100 rounded-full"><i data-lucide="x"></i></button></div>

                                </form>
                            </div>
                        </div>
                    )}
                </div>
            );
        };

        const BudgetSummary = ({ trip }) => {
            const [activities, setActivities] = useState([]);
            const [loading, setLoading] = useState(true);

            useEffect(() => {
                const fetchAll = async () => {
                    if (trip.id === 'demo-rajasthan') { setActivities(trip.itinerary.days.flatMap(d => d.activities.map(a => ({ ...a, day_number: d.dayNumber })))); setLoading(false); return; }
                    try {
                        const { data: days, error } = await supabase.from('days').select('id, day_number').eq('trip_id', trip.id);
                        if (error) throw error;
                        let all = [];
                        for (const day of days) {
                            const { data: acts } = await supabase.from('activities').select('*').eq('day_id', day.id);
                            all = [...all, ...acts.map(a => ({ ...a, day_number: day.day_number }))];
                        }
                        setActivities(all);
                    } catch (err) {} finally { setLoading(false); }
                };
                fetchAll();
            }, [trip]);

            if (loading) return <div className="flex flex-col items-center justify-center p-20"><Spinner /><p className="mt-4 font-bold text-primary animate-pulse">Calculating expenses...</p></div>;

            const total = activities.reduce((sum, a) => sum + a.cost, 0);
            const budget = trip.budget || 0;
            const percentage = Math.min(Math.round((total / budget) * 100), 100);
            const byCategory = activities.reduce((acc, a) => { acc[a.category] = (acc[a.category] || 0) + a.cost; return acc; }, {});
            const byDay = activities.reduce((acc, a) => { acc[a.day_number] = (acc[a.day_number] || 0) + a.cost; return acc; }, {});
            const maxDayCost = Math.max(...Object.values(byDay), 1);

            return (
                <div className="space-y-8 animate-in fade-in duration-500">
                    <h2 className="text-2xl font-bold text-gray-900">Budget Summary</h2>
                    {total > budget && <div className="bg-danger/10 border border-danger/20 p-4 rounded-2xl flex items-center gap-3 text-danger"><i data-lucide="alert-triangle" className="w-5 h-5"></i><p className="text-sm font-bold">Budget exceeded by ₹{(total - budget).toLocaleString()}</p></div>}
                    <div className="bg-white rounded-[2rem] p-8 shadow-md border border-gray-100 space-y-6">
                        <div className="flex justify-between items-end"><div><p className="text-xs font-bold text-gray-400 uppercase tracking-widest mb-1">Total Spent</p><p className="text-4xl font-bold text-primary">₹{total.toLocaleString()}</p></div><div className="text-right"><p className="text-xs font-bold text-gray-400 uppercase tracking-widest mb-1">Budget</p><p className="text-xl font-bold text-gray-700">₹{budget.toLocaleString()}</p></div></div>
                        <div className="h-4 bg-gray-100 rounded-full overflow-hidden"><div className={`h-full transition-all duration-1000 ${total > budget ? 'bg-danger' : percentage > 80 ? 'bg-warning' : 'bg-success'}`} style={{ width: `${percentage}%` }}></div></div>
                        <div className="flex justify-between text-xs font-bold"><span className={total > budget ? 'text-danger' : 'text-success'}>{percentage}% Used</span><span className="text-gray-400">₹{(budget - total).toLocaleString()} Remaining</span></div>
                    </div>
                    <div className="grid grid-cols-1 sm:grid-cols-2 gap-6">
                        <div className="bg-white rounded-[2rem] p-6 shadow-md border border-gray-100 space-y-4">
                            <h3 className="font-bold text-gray-900 flex items-center gap-2"><i data-lucide="pie-chart" className="w-4 h-4 text-accent"></i> Categories</h3>
                            <div className="space-y-3">{Object.entries(byCategory).map(([cat, cost]) => <div key={cat} className="flex items-center justify-between"><Badge category={cat}>{cat}</Badge><span className="font-bold text-gray-700">₹{cost.toLocaleString()}</span></div>)}</div>
                        </div>
                        <div className="bg-white rounded-[2rem] p-6 shadow-md border border-gray-100 space-y-4">
                            <h3 className="font-bold text-gray-900 flex items-center gap-2"><i data-lucide="bar-chart-2" className="w-4 h-4 text-accent"></i> Daily Spend</h3>
                            <div className="flex items-end justify-between h-32 pt-4 px-2">{Object.entries(byDay).map(([day, cost]) => (
                                <div key={day} className="flex flex-col items-center gap-2 flex-1 group relative">
                                    <div className="absolute -top-6 bg-primary text-white text-[8px] px-1.5 py-0.5 rounded opacity-0 group-hover:opacity-100 transition-all">₹{cost}</div>
                                    <div className="w-4 bg-accent/20 rounded-t-sm transition-all duration-500 hover:bg-accent" style={{ height: `${(cost / maxDayCost) * 100}%` }}></div>
                                    <span className="text-[10px] font-bold text-gray-400">D{day}</span>
                                </div>
                            ))}</div>
                        </div>
                    </div>
                </div>
            );
        };

                            <>
                                {currentScreen === 'PLANNER' && <DayPlanner trip={activeTrip} onUpdate={() => fetchTrips(user.id)} showToast={showToast} />}
                                {currentScreen === 'PACKING' && <PackingList trip={activeTrip} onUpdate={() => fetchTrips(user.id)} showToast={showToast} />}
                                {currentScreen === 'BUDGET' && <BudgetSummary trip={activeTrip} />}
                            </>
                        )}
                    </main>
                    {activeTrip && <BottomNav current={currentScreen} setScreen={setCurrentScreen} />}
                    {toast && <Toast {...toast} onClose={() => setToast(null)} />}
                </div>
            );
        }

        ReactDOM.createRoot(document.getElementById('root')).render(<App />);
    </script>
</body>
